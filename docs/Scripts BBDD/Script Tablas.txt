-- Table: "Cliente"

-- DROP TABLE "Cliente";

CREATE TABLE "Cliente"
(
  "userName" text NOT NULL,
  pass text NOT NULL,
  nombre text,
  apellido text,
  telefono text,
  email text NOT NULL,
  privilegios integer NOT NULL,
  baneado integer NOT NULL,
  tarjeta text,
  CONSTRAINT pk_cliente PRIMARY KEY ("userName")
)
WITH (
  OIDS=FALSE
);
ALTER TABLE "Cliente"
  OWNER TO postgres;
